@meta {
	name "Random Entities Support";
	description "Installs mods to support random entities in resource packs";
	authors "mcvm";
}

@install {
	if loader forgelike {
		require "optifine-support";
		finish;
	}
	if loader fabriclike {
		require "entity-texture-features";
		finish;
	}
	fail unsupported_loader;
}
